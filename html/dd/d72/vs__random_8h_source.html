46 {
47 #define NN_CONSTANT 624
48  static const unsigned int N()
49  {

52  static const int M()
53  {
54  return 397;
55  }

57  {

59  }

62  return 0x80000000UL;

66  return 0x7fffffffUL;

68  unsigned int mt[NN_CONSTANT]; /* the array for the state vector */
69 #undef NN_CONSTANT
70  unsigned int mti; /* mti==N+1 means mt[N] is not initialized */
71  /* initializes mt[N] with a seed */
72 public:
73  VSRandom( unsigned int s ) : mti( N()+1 )
74  {
75  init_genrand( s );
76  }
77  void init_genrand( unsigned int s )
78  {
79  mt[0] = s&0xffffffffUL;
80  for (mti = 1; mti < N(); mti++)
81  {
82  mt[mti] =
83  (1812433253UL*( mt[mti-1]^(mt[mti-1]>>30) )+mti);
84  /*
85  * See Knuth TAOCP Vol2. 3 rd Ed. P.106 for multiplier.
86  * In the previous versions, MSB s of the seed affect
87  * only MSBs of the array mt[].
88  * 2002/01/09 modified by Makoto Matsumoto
89  */
90  mt[mti] &= 0xffffffffUL;
91  /* for >32 bit machines */
92  }
93  }
94  /*
95  * initialize by an array with array-length
96  * init_key is the array for initializing keys
97  * key_length is its length
98  */
99  VSRandom( unsigned int init_key[], unsigned int key_length ) : mti( N()+1 )
100  {
101  unsigned int i, j, k;
102  init_genrand( 19650218UL );
103  i = 1;
104  j = 0;
105  k = (N() > key_length ? N() : key_length);
106  for (; k; k--)
107  {
108  mt[i] = ( mt[i]^( ( mt[i-1]^(mt[i-1]>>30) )*1664525UL ) )
109  +init_key[j]+j; /* non linear */
110  mt[i] &= 0xffffffffUL; /* for WORDSIZE > 32 machines */
111  i++;
112  j++;
113  if ( i >= N() )
114  {
115  mt[0] = mt[N()-1];
116  i = 1;
117  }
118  if (j >= key_length) j = 0;
119  }
120  for (k = N()-1; k; k--)
121  {
122  mt[i] = ( mt[i]^( ( mt[i-1]^(mt[i-1]>>30) )*1566083941UL ) )
123  -i; /* non linear */
124  mt[i] &= 0xffffffffUL; /* for WORDSIZE > 32 machines */
125  i++;
126  if ( i >= N() )
127  {
128  mt[0] = mt[N()-1];
129  i = 1;
130  }
131  }
132  mt[0] = 0x80000000UL; /* MSB is 1; assuring non-zero initial array */
133  }
134  /* generates a random number on [0,0xffffffff]-interval */
135  unsigned int genrand_int32( void )
136  {
137  unsigned int y;
138 
139  //The literal and the MATRIX_A had to be static_cast after C++11
140  static unsigned int mag01[2] = {static_cast<unsigned int>(0x0UL), static_cast<unsigned int>(MATRIX_A())};
141  /* mag01[x] = x * MATRIX_A for x=0,1 */
142  if ( mti >= N() )
143  {
144  /* generate N words at one time */
145  unsigned int kk;
146  if (mti == N()+1) /* if init_genrand() has not been called, */
147  init_genrand( 5489UL ); /* a default initial seed is used */
148  for (kk = 0; kk < N()-M(); kk++)
149  {
150  y = ( mt[kk]&UPPER_MASK() )|( mt[kk+1]&LOWER_MASK() );
151  mt[kk] = mt[kk+M()]^(y>>1)^mag01[y&0x1UL];
152  }
153  for (; kk < N()-1; kk++)
154  {
155  y = ( mt[kk]&UPPER_MASK() )|( mt[kk+1]&LOWER_MASK() );
156  mt[kk] = mt[kk+( M()-N() )]^(y>>1)^mag01[y&0x1UL];
157  }
158  y = ( mt[N()-1]&UPPER_MASK() )|( mt[0]&LOWER_MASK() );
159  mt[N()-1] = mt[M()-1]^(y>>1)^mag01[y&0x1UL];
160  mti = 0;
161  }
162  y = mt[mti++];
163  /* Tempering */
164  y ^= (y>>11);
165  y ^= (y<<7)&0x9d2c5680UL;
166  y ^= (y<<15)&0xefc60000UL;
167  y ^= (y>>18);
168  return y;
169  }
170  /* generates a random number on [0,0x7fffffff]-interval */
171  int genrand_int31( void )

174  }
