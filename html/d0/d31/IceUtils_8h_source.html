223  // Compute implicit coords from an index:
224  // The idea is to get back 2D coords from a 1D index.
225  // For example:
226  //
227  // 0 1 2 ... nbu-1
228  // nbu nbu+1 i ...
229  //
230  // We have i, we're looking for the equivalent (u=2, v=1) location.
231  // i = u + v*nbu
232  // <=> i/nbu = u/nbu + v
233  // Since 0 <= u < nbu, u/nbu = 0 (integer)
234  // Hence: v = i/nbu
235  // Then we simply put it back in the original equation to compute u = i - v*nbu
236  inline_ void Compute2DCoords(udword& u, udword& v, udword i, udword nbu)
237  {
238  v = i / nbu;
239  u = i - (v * nbu);
240  }
