55 {
56  int i;
57  size_t rode = f.Read( b, sizeof (*b)*n );
58 #ifndef NATURAL_ENDIANNESS

60  ( (unsigned int*) b )[i] = le32_to_cpu( ( (unsigned int*) b )[i] );

62  return (rode > 0) ? ( rode/sizeof (*b) ) : rode;
