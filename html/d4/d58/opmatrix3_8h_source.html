122  {
123  csMatrix3 C(

125  -(m21*m33 - m23*m31), (m11*m33 - m13*m31), -(m11*m23 - m13*m21),
126  (m21*m32 - m22*m31), -(m11*m32 - m12*m31), (m11*m22 - m12*m21) );
127  float s = (float)1./(m11*C.m11 + m12*C.m21 + m13*C.m31);
